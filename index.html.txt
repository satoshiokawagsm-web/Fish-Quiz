<script>

// ★ここはあなたの元データそのまま使う（ズレ防止）
const fishList = [
    { name: "アカマツカサ", image: "fish1.jpg" },
    { name: "ハリセンボン", image: "fish2.jpg" },
    { name: "アオウミガメ", image: "fish3.jpg" },
    { name: "クマノミ", image: "fish4.jpg" },
    { name: "ムラサメモンガラ", image: "fish5.jpg" },
    { name: "ゴマモンガラ", image: "fish6.jpg" },
    { name: "カクレクマノミ", image: "fish7.jpg" },
    { name: "ロクセンスズメダイ", image: "fish8.jpg" },
    { name: "ナンヨウハギ", image: "fish9.jpg" },
    { name: "ハタタテハゼ", image: "fish10.jpg" }
];

// 🔊 音
const correctSound = new Audio("https://assets.mixkit.co/active_storage/sfx/2013/2013-preview.mp3");
const wrongSound = new Audio("https://assets.mixkit.co/active_storage/sfx/2003/2003-preview.mp3");

// =====================
// タップクイズ
// =====================
let current;
let showName = false;

function startNormal(){
    document.getElementById("normal").classList.remove("hidden");
    document.getElementById("quiz").classList.add("hidden");
    nextNormal();
}

function nextNormal(){
    current = fishList[Math.floor(Math.random()*fishList.length)];
    document.getElementById("img-normal").src = current.image;
    document.getElementById("name-normal").innerText = "";
    showName = false;
}

document.getElementById("normal").onclick = () => {
    if(!showName){
        document.getElementById("name-normal").innerText = current.name;
        correctSound.play();
        showName = true;
    }else{
        nextNormal();
    }
};

// =====================
// 4択クイズ（ズレない版）
// =====================
let quizCurrent;

function startQuiz(){
    document.getElementById("normal").classList.add("hidden");
    document.getElementById("quiz").classList.remove("hidden");
    nextQuiz();
}

function nextQuiz(){

    quizCurrent = fishList[Math.floor(Math.random()*fishList.length)];

    let choices = [quizCurrent];

    while(choices.length < 4){
        let r = fishList[Math.floor(Math.random()*fishList.length)];
        if(!choices.includes(r)){
            choices.push(r);
        }
    }

    choices.sort(()=>Math.random()-0.5);

    document.getElementById("img-quiz").src = quizCurrent.image;

    const c = document.getElementById("choices");
    c.innerHTML = "";

    choices.forEach(item=>{
        const btn = document.createElement("button");
        btn.innerText = item.name;
        btn.className = "choice";

        btn.onclick = ()=>{
            if(item === quizCurrent){
                correctSound.play();
                alert("正解！");
            }else{
                wrongSound.play();
                alert("不正解！");
            }
            nextQuiz();
        };

        c.appendChild(btn);
    });
}

// 初期表示
startNormal();

</script>
